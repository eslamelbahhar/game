                introContainer.style.display = 'none'; resultsContainer.style.display = 'none';

                gameContainer.style.display = 'block';

            function loadQuestion() {

                if (currentQuestionIndex < questions.length) {

                    const questionData = questions[currentQuestionIndex];

                    questionsContainer.innerHTML = ''; 


                    const card = document.createElement('div'); card.className = 'card question-card';

                        const scenarioDiv = document.createElement('div'); scenarioDiv.className = 'scenario-card';

                        scenarioDiv.innerHTML = `<strong>Scenario:</strong><p style="margin-bottom:0;">${questionData.scenario}</p>`;

                        card.appendChild(scenarioDiv);

                    }

                    const questionTitle = document.createElement('h3'); questionTitle.className = 'mb-10';

            function showResults() {

                feedbackTextEl.innerHTML = `<p>${resultMessage}</p>`;
